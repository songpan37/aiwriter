<project_specification>
  <project_name>AI写作辅助系统</project_name>

  <overview>
    构建一个功能完善的AI写作辅助系统，帮助作者高效管理创作流程，提升作品质量。系统包含作品管理、作品创作、AI优化、一键发布四大核心模块。前端采用React+TypeScript+Vite构建现代化单页应用，后端采用Go语言提供高性能API服务。界面布局清晰：顶部为全局信息栏（作品切换、通知、用户信息），左侧为导航菜单，右侧为内容展示区，确保操作直观高效。
  </overview>

  <technology_stack>
    <frontend>
      <framework>React 18 + TypeScript + Vite</framework>
      <state_management>React Context + useReducer / Zustand</state_management>
      <routing>React Router v6</routing>
      <ui_components>Ant Design / 自定义组件</ui_components>
      <styling>CSS Modules / Tailwind CSS</styling>
      <http_client>axios</http_client>
      <diff_view>react-diff-viewer / 自定义文本对比组件</diff_view>
    </frontend>
    <backend>
      <language>Go 1.20+</language>
      <framework>Gin / Echo</framework>
      <database>MySQL / PostgreSQL</database>
      <orm>GORM</orm>
      <authentication>JWT</authentication>
      <ai_integration>OpenAI API / 自研模型接口</ai_integration>
    </backend>
    <communication>
      <api>RESTful JSON API</api>
      <file_upload>multipart/form-data</file_upload>
    </communication>
  </technology_stack>

  <prerequisites>
    <environment>
      - 前端开发环境需Node.js 16+，包管理器pnpm/npm
      - 后端开发环境需Go 1.20+，并配置好数据库连接
      - AI优化模块需配置第三方AI服务API密钥
    </environment>
  </prerequisites>

  <core_features>
    <!-- 作品列表模块 -->
    <feature_group name="作品列表管理">
      <feature>作品列表展示：以网格或列表形式展示所有作品，每项包含封面图、书名、章节数、总字数、最后修改时间。</feature>
      <feature>作品筛选与排序：支持按作品类别、修改时间筛选和排序。</feature>
      <feature>作品详情入口：点击作品卡片跳转至作品创作页面的基本信息子页。</feature>
      <feature>作品删除：提供删除按钮，点击后二次确认，从数据库中软删除或硬删除。</feature>
      <feature>新增作品向导：
        - 弹出式向导或页面内表单
        - 作品类别下拉框（可预定义类别如玄幻、都市等）
        - 同类作品参考多项选择（展示已有作品供参考）
        - 基本信息输入（书名、作者、简介等）
        - 每卷章节数输入（期望每卷包含多少章节）
        - 每章节字数输入（期望每章节字数）
        - 确认后调用创建接口，生成作品基础结构
      </feature>
    </feature_group>

    <!-- 作品创作模块 -->
    <feature_group name="作品创作">
      <feature>左侧菜单栏动态导航：
        - 根菜单：基本信息、分卷信息、章节信息，支持展开/折叠
        - 基本信息子菜单：点击后右侧展示基本信息编辑表单
        - 分卷信息子菜单：点击后展示分卷列表，可进一步展开每卷的详细子项
        - 章节信息子菜单：按卷分组展示章节列表，可展开每章的详细子项
      </feature>
      <feature>基本信息编辑：
        - 字段：书名（文本）、故事梗概（多行文本）、时空设定（文本）、力量体系（文本）、主要人物（富文本/列表）、封面图片（上传）
        - 支持实时保存或手动保存
      </feature>
      <feature>分卷信息管理：
        - 分卷列表展示（卷名、章节数、字数）
        - 点击某卷进入详细子页面，展示该卷的：
          - 卷名、当前章节数、字数（自动计算）
          - 主要情节概述（富文本）
          - 出场人物（多选标签）
          - 主要的情节单元（可添加多个单元，每个单元包含多个章节）
          - 关键人物与关系变化（文本）
          - 主角目标与进展（文本）
          - 冲突与转折（文本）
          - 伏笔与后续钩子（文本）
          - 每一章节的情节概述（列表）
        - 支持新增、编辑、删除分卷
      </feature>
      <feature>章节信息管理：
        - 按卷分组展示章节列表（章节名、剧情梗概）
        - 点击某章节进入详情编辑：
          - 章节名、剧情梗概、故事所在时间空间、出场人物（多选）
          - 场景列表：可添加多个场景，每个场景包含：
            - 开端（文本）
            - 发展与冲突（文本）
            - 转折（文本）
            - 结果（文本）
            - 风格（下拉/标签）
        - 支持新增、编辑、删除章节
      </feature>
    </feature_group>

    <!-- AI优化模块 -->
    <feature_group name="AI优化流程">
      <feature>优化步骤列表：默认包含四个步骤——信息认知时序优化、人物对话优化、滥用表达优化、抽象设定说明优化，按顺序排列。</feature>
      <feature>步骤详情展示：点击某步骤，右侧展示该步骤的详细内容，包括：
        - 审阅提示词（只读）
        - 审阅结论（AI生成的分析结论）
        - 原文优化提示词（模板+实际填充）
        - 原文内容（输入文本）
        - 优化后的原文内容（AI生成）
        - 优化前后文本对比视图（高亮差异，支持左右对照或行内对比）
      </feature>
      <feature>添加自定义步骤：
        - 弹出表单：输入审阅提示词模板、原文优化提示模板
        - 选择插入位置（在某个步骤之后）
        - 确认后新步骤加入列表，并可立即使用
      </feature>
      <feature>步骤执行控制：
        - 支持手动触发每个步骤的AI优化（调用AI接口）
        - 上一步的输出作为下一步的输入自动传递
        - 可重置某步骤的优化结果
      </feature>
      <feature>优化历史记录：保存每次优化的版本，支持回滚查看。</feature>
    </feature_group>

    <!-- 一键发布模块 -->
    <feature_group name="一键发布">
      <feature>作品选择：下拉框选择要发布的作品，自动加载该作品的章节列表。</feature>
      <feature>章节选择：多选框选择要发布的章节，提供全选/反选功能。</feature>
      <feature>字数重划分：
        - 输入自定义字数（如2000字/章）
        - 根据该字数将选中章节的原文内容重新划分章节，生成新的章节列表
        - 支持预览重新划分后的章节结构
      </feature>
      <feature>章节名自动生成：AI根据重新划分后的章节内容生成新的章节名，可手动编辑。</feature>
      <feature>发布目标平台：选择要发布的平台（番茄、起点等），可配置各平台发布参数（如分类、标签）。</feature>
      <feature>发布执行：点击发布按钮，调用后端发布接口，将作品内容推送至目标平台，并反馈发布结果。</feature>
    </feature_group>

  <key_interactions>
    <create_work_wizard>
      1. 用户点击“新增作品”按钮
      2. 弹出向导对话框，第一步选择作品类别
      3. 第二步选择同类作品参考（多选）
      4. 第三步填写作品信息（书名、简介等）以及每卷章节数、每章字数
      5. 点击确认，前端组装数据，POST /api/works 创建作品，成功后跳转至作品创作页
    </create_work_wizard>
    <ai_optimization_flow>
      1. 用户选择某步骤，点击“执行优化”
      2. 前端将当前原文（来自上一优化步骤输出或原始内容）和步骤模板发送至后端，后端调用AI服务
      3. 后端返回优化后文本和审阅结论，前端展示差异对比
      4. 用户可继续下一步优化，数据自动传递
    </ai_optimization_flow>
    <publish_flow>
      1. 用户选择作品和章节，输入目标字数，点击“重新划分”
      2. 前端将章节内容发送至后端，后端调用AI（或本地算法）重新分割，并生成新章节名
      3. 前端展示新章节列表，用户可编辑章节名
      4. 用户选择平台，点击“发布”，后端执行异步发布任务，前端轮询状态更新
    </publish_flow>
  </key_interactions>
