<project_specification>
  <project_name>AI写作辅助系统 - 小说创作平台</project_name>

  <overview>
    构建一个专业的AI写作辅助系统，帮助作者进行小说创作和管理。系统采用前后端分离架构，前端使用React+TypeScript+Vite，后端使用Go。提供作品管理、创作详情（分卷、章节、AI优化）、自动创作流程控制以及一键发布到第三方平台等功能。产品目标是为作者提供从构思、创作到发布的全流程智能化辅助，提升创作效率与质量。
  </overview>

  <technology_stack>
    <frontend>
      <framework>React 18 + TypeScript + Vite</framework>
      <state_management>Zustand 或 Context API 用于全局状态</state_management>
      <routing>React Router v6 实现页面导航</routing>
      <ui_library>Ant Design 或 Semi Design 用于基础组件</ui_library>
      <styling>CSS Modules 或 Tailwind CSS（根据团队偏好）</styling>
      <http_client>Axios 与后端交互</http_client>
      <diff_view>react-diff-viewer 或类似库展示文本对比</diff_view>
      <port>默认启动端口 3000（可配置）</port>
    </frontend>
    <backend>
      <language>Go 1.20+</language>
      <framework>Gin 或 Echo 用于RESTful API</framework>
      <database>MySQL（关系数据） + Redis（缓存/会话）</database>
      <orm>GORM 或 sqlx</orm>
      <authentication>JWT（Access Token + Refresh Token）</authentication>
      <ai_integration>对接大语言模型API（如OpenAI、自研模型）用于内容生成与优化</ai_integration>
      <file_storage>本地或对象存储（MinIO/AWS S3）用于封面图片</file_storage>
      <port>默认启动端口 8080</port>
    </backend>
    <communication>
      <api>RESTful JSON over HTTP</api>
      <websocket>可选，用于实时进度推送（创作步骤）</websocket>
    </communication>
  </technology_stack>

  <prerequisites>
    <environment_setup>
      - 前端项目已通过Vite初始化，包含TypeScript配置。
      - 后端Go项目已初始化，包含依赖管理。
      - 数据库MySQL已安装并创建空数据库。
      - Redis已安装用于缓存和会话存储。
      - 环境变量管理：前端通过.env文件，后端通过配置文件或系统环境变量。
    </environment_setup>
  </prerequisites>

  <core_features>
    <!-- 全局布局与公共模块 -->
    <global_layout>
      - 顶端公共栏：作品下拉列表（切换当前操作作品）、系统通知图标、用户信息下拉（头像、昵称、退出登录）。
      - 左侧菜单栏：根据当前选择的作品，展示导航菜单（作品详情入口、发布管理等）；菜单可折叠/展开。
      - 右侧内容区域：根据左侧菜单选择展示对应内容。
      - 底部固定操作栏：显示“开始创作”按钮，点击后切换为“停止创作”；当创作进行时显示当前步骤和进度。
    </global_layout>

    <!-- 作品列表管理 -->
    <workspace_management>
      - 作品列表页：展示当前用户所有作品，以卡片或表格形式呈现，每项包含：封面图片、书名、章节数、总字数、最后修改时间。
      - 操作：点击作品卡片进入作品详情页；点击删除按钮弹出确认框，确认后删除作品。
      - 新增作品：点击“新增”按钮弹出创作向导。
        - 第一步：选择作品类别（从预定义类别下拉选择）。
        - 第二步：多项选择同类作品参考（从已有作品中多选，作为创作参考）。
        - 第三步：填写基本信息（书名、作者署名、一句话简介等）。
        - 第四步：输入期望每卷章节数、每章节字数。
        - 确认后，系统创建空白作品结构（分卷、章节占位），并可根据参考作品预填充部分设定。
    </workspace_management>

    <!-- 作品详情 - 基础信息 -->
    <work_detail_basic>
      - 左侧菜单展开“基本信息”，右侧展示基本信息编辑表单，包含：
        - 书名（文本输入）
        - 故事梗概（多行文本）
        - 时空设定（文本域）
        - 力量体系（文本域，支持富文本）
        - 主要人物（可添加多个，每个包含姓名、角色描述、性格等）
        - 封面图片（上传图片，预览）
      - 所有字段可编辑并保存。
    </work_detail_basic>

    <!-- 作品详情 - 分卷信息 -->
    <work_detail_volume>
      - 左侧菜单展开“分卷信息”，右侧展示分卷列表，每个分卷可展开查看详情。
      - 分卷数据结构：卷名、当前章节数、总字数、主要情节概述、出场人物（列表）、主要情节单元（包含多个章节）、关键人物与关系变化、主角目标与进展、冲突与转折、伏笔与后续钩子、每一章节的情节概述（列表）。
      - 操作：新增分卷、编辑分卷、删除分卷；在分卷内可管理章节（见章节信息）。
    </work_detail_volume>

    <!-- 作品详情 - 章节信息 -->
    <work_detail_chapter>
      - 左侧菜单展开“章节信息”，右侧按卷分组展示章节列表。
      - 每个章节条目包含：章节名、剧情梗概、故事时间空间、出场人物、场景列表。
      - 场景列表可展开，每个场景包含：开端、发展与冲突、转折、结果、风格。
      - 操作：新增章节、编辑章节、删除章节；调整章节顺序（拖拽）；支持批量导入/导出。
    </work_detail_chapter>

    <!-- AI优化 -->
    <ai_optimization>
      - 左侧菜单选择“AI优化”，右侧展示优化工作流界面。
      - 默认优化步骤（可配置）：信息认知时序优化、人物对话优化、滥用表达优化、抽象设定说明优化。
      - 每个优化步骤卡片展示：
        - 审阅提示词（可编辑）
        - 审阅结论（AI返回的分析结论）
        - 原文优化提示词（可编辑）
        - 原文内容（待优化的文本，可来自选中的章节或用户输入）
        - 优化后的原文内容（AI生成结果）
        - 文本对比区域：并排显示原文和优化后内容，高亮差异。
      - 步骤执行顺序：串行，上一步的输出作为下一步的输入（默认将优化后内容传递给下一步作为原文）。
      - 用户可以：
        - 添加新优化步骤：输入审阅提示词模板、原文优化提示模板，选择插入到某步骤之后。
        - 删除步骤（非默认步骤）。
        - 调整步骤顺序（拖拽）。
        - 单独运行某一步（基于当前输入）或运行全部。
        - 保存优化结果为最终版本。
    </ai_optimization>

    <!-- 自动创作控制 -->
    <auto_creation>
      - 底部固定栏显示“开始创作”按钮。
      - 点击“开始创作”后，按钮变为“停止创作”，系统按预设逻辑（如从当前章节开始，逐步生成内容）自动进行创作。
      - 创作过程中，底部显示当前执行的步骤（如“生成第3章-场景2”）、进度条或百分比。
      - 用户可随时点击“停止创作”暂停流程，再次点击“开始创作”可继续（需记录断点）。
      - 创作进度需持久化，支持页面刷新后恢复。
    </auto_creation>

    <!-- 作品发布 -->
    <work_publish>
      - 左侧菜单“发布”入口，进入发布配置页面。
      - 选择发布平台：番茄、起点等（可配置列表）。
      - 选择要发布的作品（默认为当前作品）。
      - 选择要发布的章节（可全选或单选）。
      - 自动填充上次发布记录：上次发布的平台、章节范围、起始章节号等。
      - 发布选项：
        - 按AI自动生成的原始章节发布。
        - 按自定义字数重新划分章节：输入目标字数，系统自动合并/拆分章节，并重新生成章节名，然后发布。
      - 番茄平台增值服务：
        - 一键生成多书名+封面：基于作品内容，AI生成多个备选书名和对应封面图，帮助作者进行多书名实验。
      - 发布前预览发布内容（模拟），确认后执行发布，记录发布日志。
    </work_publish>

    <!-- 辅助功能 -->
    <auxiliary>
      - 用户认证与权限管理：登录/注册、个人资料修改。
      - 作品类别管理（后台管理员可配置）。
      - 系统通知：展示平台公告、系统消息。
      - 操作日志：记录用户关键操作。
    </auxiliary>
  </core_features>

  <database_schema>
    <tables>
      <users>
        - id (uuid, primary key)
        - username, email, password_hash
        - avatar_url, created_at, updated_at, last_login
        - preferences (json: theme, notification_settings, etc.)
      </users>

      <categories>
        - id (serial)
        - name (string), description, created_at
      </categories>

      <works>
        - id (uuid)
        - user_id (references users)
        - category_id (references categories)
        - title, author_name, cover_image_url
        - synopsis (text), setting (text), power_system (text)
        - total_chapters, total_words
        - created_at, updated_at, last_modified_at
        - status (draft/active/archived)
      </works>

      <reference_works>  <!-- 同类作品参考关联表 -->
        - id (serial)
        - work_id (references works)
        - reference_work_id (references works)  -- 被参考的作品
      </reference_works>

      <volumes>
        - id (uuid)
        - work_id (references works)
        - volume_number, volume_name
        - chapter_count, word_count
        - plot_overview (text), characters (jsonb list of names)
        - plot_units (jsonb list of plot unit summaries)
        - key_relationships (text)
        - protagonist_goal (text)
        - conflicts (text)
        - foreshadowing (text)
        - chapter_overviews (jsonb list of strings)
        - created_at, updated_at
      </volumes>

      <chapters>
        - id (uuid)
        - volume_id (references volumes)
        - chapter_number, chapter_title
        - synopsis (text)
        - time_space (string)
        - characters (jsonb list of names)
        - scenes (jsonb array of objects: {opening, development_conflict, twist, outcome, style})
        - word_count
        - created_at, updated_at
      </chapters>

      <ai_optimization_steps>
        - id (serial)
        - user_id (references users)  -- 自定义步骤属于用户
        - name (string)  -- 步骤名称
        - review_prompt_template (text)
        - optimize_prompt_template (text)
        - is_default (boolean)
        - created_at
      </ai_optimization_steps>

      <optimization_tasks>  <!-- 记录每次优化操作，可用于历史追溯 -->
        - id (uuid)
        - work_id (references works)
        - step_id (references ai_optimization_steps)
        - input_text (text)
        - review_output (text)
        - optimized_text (text)
        - created_at
      </optimization_tasks>

      <auto_creation_progress>  <!-- 自动创作断点记录 -->
        - id (uuid)
        - work_id (references works)
        - current_step_type (string: volume/chapter/scene)
        - current_step_id (uuid)
        - progress_percentage (int)
        - updated_at
      </auto_creation_progress>

      <publish_platforms>
        - id (serial)
        - name (string), display_name, api_endpoint, is_active
      </publish_platforms>

      <publish_records>
        - id (uuid)
        - work_id (references works)
        - platform_id (references publish_platforms)
        - published_chapters (jsonb array of chapter ids)
        - start_chapter (int)  -- 从第几章开始发布（用于续发）
        - custom_split_words (int, nullable)  -- 若按字数重分，记录字数
        - created_at
      </publish_records>

      <notifications>
        - id (uuid)
        - user_id (references users)
        - title, content, is_read
        - created_at
      </notifications>

      <operation_logs>
        - id (uuid)
        - user_id, action, entity_type, entity_id, details (json)
        - created_at
      </operation_logs>
    </tables>
  </database_schema>

  <key_interactions>
    <work_creation_wizard>
      1. 用户点击“新增”按钮，打开向导弹窗。
      2. 第一步选择类别，第二步多选参考作品（可选），第三步输入基本信息和字数目标。
      3. 确认后系统创建作品，并根据参考作品（如果有）初始化部分设定（如人物、力量体系等），然后跳转到作品详情页。
    </work_creation_wizard>

    <ai_optimization_flow>
      1. 用户进入AI优化页面，看到默认步骤列表。
      2. 用户可在输入区域粘贴原文，或从章节选择器选取。
      3. 点击“运行所有步骤”，系统按顺序调用AI接口，每步完成时实时更新卡片内容，并展示差异对比。
      4. 用户可点击单步“运行”，仅执行该步。
      5. 用户可添加自定义步骤：填写两个提示模板，选择插入位置，确认后新步骤出现在列表中。
      6. 用户可调整步骤顺序（拖拽）。
      7. 最终优化结果可保存到章节内容。
    </ai_optimization_flow>

    <auto_creation_control>
      1. 用户在作品详情页，点击底部“开始创作”。
      2. 系统根据当前作品状态（如未开始的章节）开始生成内容，逐步填充章节和场景。
      3. 底部进度条实时更新，显示当前步骤。
      4. 用户可随时点击“停止创作”中断，系统保存断点。
      5. 再次点击“开始创作”从断点继续。
    </auto_creation_control>

    <publish_flow>
      1. 用户进入发布页面，默认展示上次发布记录（如果有）。
      2. 选择平台，作品（默认当前），章节范围。
      3. 若选择“按字数重分”，输入目标字数，系统即时预览新章节划分。
      4. 若平台为番茄，点击“生成多书名+封面”，AI生成多个方案展示，用户可选择保存或直接发布。
      5. 点击“发布”，系统执行发布操作，完成后记录日志并提示成功。
    </publish_flow>
  </key_interactions>
